 SELECT name FROM SONGS

 SELECT name FROM SONGS
    WHERE
    (
        danceability > 0.75 AND energy > 0.75 AND valence > 0.75
    );

    WHERE
    (
        artist_id = (
            SELECT id
            FROM artists
            WHERE name = "Post Malone"
        )
    );

            SELECT id

        )
    );
